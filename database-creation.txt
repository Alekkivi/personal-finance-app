CREATE DATABASE personal_finance_app;
USE personal_finance_app;

CREATE TABLE expenses_monthly
(
    id               INT unsigned NOT NULL AUTO_INCREMENT,
    username            VARCHAR(150) NOT NULL,
    expense_name     VARCHAR(150) NOT NULL,
    amount           INT unsigned NOT NULL,
    expiration       DATE NOT NULL,
    accounted_for    DATE NOT NULL,
    PRIMARY KEY      (id)
);

CREATE TABLE income_monthly
(
    id                 INT unsigned NOT NULL AUTO_INCREMENT,
    username              VARCHAR(150) NOT NULL,
    income_name        VARCHAR(150) NOT NULL,
    amount             INT unsigned NOT NULL,
    receiving_date     DATE NOT NULL,
    accounted_for      DATE NOT NULL,
    PRIMARY KEY        (id)
);

CREATE TABLE user_info
(
    id                      INT unsigned NOT NULL AUTO_INCREMENT,
    username       VARCHAR(150) NOT NULL,
    savings_base            INT unsigned NOT NULL,
    debt                    INT unsigned,
    last_login              DATE NOT NULL,
    
    PRIMARY KEY              (id)
);

CREATE TABLE earned_wage
(
    id               INT unsigned NOT NULL AUTO_INCREMENT,
    username         VARCHAR(150) NOT NULL,
    month            VARCHAR(150) NOT NULL,
    wage_ammout      INT unsigned, NOT NULL,
    yearly_tax       INT unsigned NOT NULL,
    PRIMARY KEY     (id)
);

CREATE TABLE kela_limits
(
    id               I      NT unsigned NOT NULL AUTO_INCREMENT,
    username                VARCHAR(150) NOT NULL,
    start_date              DATE NOT NULL,
    end_date                DATE NOT NULL,
    time_period_limit       INT unsigned NOT NULL,
    PRIMARY KEY             (id)
);

CREATE TABLE user_credentials
(
    id               INT unsigned NOT NULL AUTO_INCREMENT,
    username         VARCHAR(150) NOT NULL,
    password_hash    VARCHAR(150) NOT NULL,
    PRIMARY KEY     (id)
);
